PARTE 1 _ CREAZIONE DATABASE E COLLEZIONE
Script: 1_crea_database.js
Descrizione: Crea il database cartelle_cliniche e la collezione pazienti con validazione dello schema dei documenti
------------------------------------------------------------------------------------------------------------------------

//seleziona/crea il database cartelle_cliniche
use cartelle_cliniche;

//crea la collezione pazienti con validazione json schema
db.createCollection("pazienti", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["nome", "ricoveri"],
      properties: {
        nome: {
          bsonType: "string",
          description: "Nome completo del paziente - obbligatorio"
        },
        ricoveri: {
          bsonType: "array",
          description: "Lista dei ricoveri del paziente - obbligatorio",
          items: {
            bsonType: "object",
            required: ["data", "diagnosi", "anamnesi", "prognosi", "data_dimissione"],
            properties: {
              data: {
                bsonType: "date",
                description: "Data di ricovero (ISODate) - obbligatorio"
              },
              diagnosi: {
                bsonType: "string",
                description: "Diagnosi medica - obbligatorio"
              },
              anamnesi: {
                bsonType: "string",
                description: "Anamnesi del paziente - obbligatorio"
              },
              prognosi: {
                bsonType: "string",
                description: "Prognosi medica - obbligatorio"
              },
              data_dimissione: {
                bsonType: "date",
                description: "Data di dimissione (ISODate) - obbligatorio"
              }
            }
          }
        }
      }
    }
  },
  validationLevel: "moderate",
  validationAction: "warn"
});

//crea un indice sul campo nome per velocizzare le ricerche per paziente
db.pazienti.createIndex({ nome: 1 });

print(">>>>> database cartelle_cliniche e collezione pazienti creati");

PARTE 3 _ ANALISI DEI TEMPI DI RICOVERO
Script: 3_analisi_tempi.js
Descrizione: Calcola il tempo medio di ricovero in giorni per ciascun paziente e la media globale su tutti i ricoveri. I risultati vengono mostrati in console
-------------------------------------------------------------------------------------------------------------------------------------------------
use cartelle_cliniche;

//pipeline di aggregazione
//$unwind srotola l'array ricoveri (un documento per ogni ricovero)
//$project calcola la durata in giorni per ogni singolo ricovero
//$group raggruppa per paziente e calcola la media individuale
//$group calcola la media globale su tutti i pazienti
//$project formatta i valori decimali a 2 cifre

var pipeline = [

  //espansione array ricoveri
  {
    $unwind: "$ricoveri"
  },

  //calcolo durata di ogni singolo ricovero in giorni
  {
    $project: {
      nome: 1,
      durata_giorni: {
        $divide: [
          { $subtract: ["$ricoveri.data_dimissione", "$ricoveri.data"] },
          1000 * 60 * 60 * 24 //millisecondi in un giorno
        ]
      }
    }
  },

  //raggruppamento per paziente e calcolo media individuale
  {
    $group: {
      _id: "$nome",
      media_giorni_paziente: { $avg: "$durata_giorni" },
      totale_ricoveri: { $sum: 1 },
      totale_giorni: { $sum: "$durata_giorni" }
    }
  },

  //calcolo media globale su tutti i pazienti
  {
    $group: {
      _id: null,
      media_globale_giorni: { $avg: "$media_giorni_paziente" },
      pazienti: {
        $push: {
          nome: "$_id",
          media_giorni: "$media_giorni_paziente",
          totale_ricoveri: "$totale_ricoveri",
          totale_giorni: "$totale_giorni"
        }
      }
    }
  },

  //formattazione valori decimali
  {
    $project: {
      _id: 0,
      media_globale_giorni: { $round: ["$media_globale_giorni", 2] },
      pazienti: {
        $map: {
          input: "$pazienti",
          as: "p",
          in: {
            nome: "$$p.nome",
            media_giorni: { $round: ["$$p.media_giorni", 2] },
            totale_ricoveri: "$$p.totale_ricoveri",
            totale_giorni: { $round: ["$$p.totale_giorni", 2] }
          }
        }
      }
    }
  }

];



//esecuzione e stampa risultati

var risultato = db.pazienti.aggregate(pipeline).toArray();

if (risultato.length === 0) {
  print("Nessun dato trovato. Assicurarsi di aver eseguito prima lo script 2");
} else {

  var report = risultato[0];

  print("REPORT TEMPI MEDI DI RICOVERO");

  //ordinamento pazienti per nome
  report.pazienti.sort(function(a, b) {
    return a.nome.localeCompare(b.nome);
  });

  //stampa dati per ogni paziente
  report.pazienti.forEach(function(p) {
    print("  Paziente: " + p.nome);
    print("  Ricoveri totali: " + p.totale_ricoveri);
    print("  Giorni totali: " + p.totale_giorni + " gg");
    print("  Media ricovero: " + p.media_giorni + " gg");
    print("  ────────────────────────────────────────────────────────");
  });

  print("");
  print("MEDIA GLOBALE DI RICOVERO (tutti i pazienti): "
        + report.media_globale_giorni + " giorni ***");
}







PARTE 4 _ QUERY DI SUPPORTO E VERIFICA
Script: 4_query_verifica.js
Descrizione: è uno script aggiuntivo con query utili per verificare i dati inseriti e ottenere ulteriori insight
------------------------------------------------------------------------------------------------------------------

use cartelle_cliniche;

//query 1__elenco di tutti i pazienti con numero di ricoveri
print("Pazienti e numero di ricoveri");
db.pazienti.aggregate([
  {
    $project: {
      _id: 0,
      nome: 1,
      numero_ricoveri: { $size: "$ricoveri" }
    }
  },
  { $sort: { nome: 1 } }
]).forEach(function(doc) {
  print(" " + doc.nome + " --> " + doc.numero_ricoveri + " ricovero/i");
});

print("");

//query 2__ricovero più lungo in assoluto
print("Ricovero più lungo");
db.pazienti.aggregate([
  { $unwind: "$ricoveri" },
  {
    $project: {
      _id: 0,
      nome: 1,
      diagnosi: "$ricoveri.diagnosi",
      data_ricovero:"$ricoveri.data",
      data_dimissione: "$ricoveri.data_dimissione",
      durata_giorni: {
        $round: [{
          $divide: [
            { $subtract: ["$ricoveri.data_dimissione", "$ricoveri.data"] },
            86400000
          ]
        }, 0]
      }
    }
  },
  { $sort: { durata_giorni: -1 } },
  { $limit: 1 }
]).forEach(function(doc) {
  print("Paziente: " + doc.nome);
  print("Diagnosi: " + doc.diagnosi);
  print("Durata: " + doc.durata_giorni + " giorni");
});

print("");

//query 3__ricovero più breve in assoluto
print("Ricovero più breve");
db.pazienti.aggregate([
  { $unwind: "$ricoveri" },
  {
    $project: {
      _id: 0,
      nome: 1,
      diagnosi: "$ricoveri.diagnosi",
      durata_giorni: {
        $round: [{
          $divide: [
            { $subtract: ["$ricoveri.data_dimissione", "$ricoveri.data"] },
            86400000
          ]
        }, 0]
      }
    }
  },
  { $sort: { durata_giorni: 1 } },
  { $limit: 1 }
]).forEach(function(doc) {
  print("Paziente: " + doc.nome);
  print("Diagnosi: " + doc.diagnosi);
  print("Durata: " + doc.durata_giorni + " giorni");
});

print("");

//uery 4_numero totale ricoveri nel database
print("Statistiche generali");
var totaleRicoveri = db.pazienti.aggregate([
  { $project: { n: { $size: "$ricoveri" } } },
  { $group: { _id: null, totale: { $sum: "$n" } } }
]).toArray()[0].totale;

var totalePazienti = db.pazienti.countDocuments();

print("Pazienti nel database: " + totalePazienti);
print("Ricoveri totali: " + totaleRicoveri);
print("Media ricoveri/paziente: " +
      (totaleRicoveri / totalePazienti).toFixed(2));
